Explore os conceitos de Composição e Associação com exemplos interativos em JavaScript/TypeScript.

Composição

Relacionamento "tem um" onde o objeto composto possui outro objeto. A parte não existe sem o todo.

Na composição, o ciclo de vida dos objetos está intimamente ligado. Se o objeto principal for destruído, seus componentes também serão destruídos.

Associação

Relacionamento mais flexível onde objetos podem existir independentemente. Uma classe "usa" ou "conhece" outra.

Na associação, os objetos têm ciclos de vida independentes. Eles podem existir separadamente e se relacionar de forma dinâmica.

Endereco.js
class Endereco {
  #logradouro;
  #numero;
  #bairro;
  #cidade;
  #estado;
  #cep;

  constructor(logradouro, numero, bairro, cidade, estado, cep) {
    this.#logradouro = logradouro;
    this.#numero = numero;
    this.#bairro = bairro;
    this.#cidade = cidade;
    this.#estado = estado;
    this.#cep = cep;
  }

  toString() {
    return `${this.#logradouro}, ${this.#numero} - ${this.#bairro}`;
  }
}
Pessoa.js
class Pessoa {
  #nome;
  #cpf;
  #endereco; // Composição: Pessoa "possui" Endereco

  constructor(nome, cpf, endereco) {
    this.#nome = nome;
    this.#cpf = cpf;
    this.#endereco = endereco; // Objeto Endereco
  }

  get nome() { return this.#nome; }
  get cpf() { return this.#cpf; }
  get endereco() { return this.#endereco; }
}

Empresa & Departamentos Associação

Diagrama de Classes

1..* (tem muitos)

Departamento.js
class Departamento {
  #nome;

  constructor(nome) {
    this.#nome = nome;
  }

  get nome() { return this.#nome; }
}
Empresa.js
class Empresa {
  #nome;
  #departamentos; // Associação: Empresa "tem" Departamentos

  constructor(nome) {
    this.#nome = nome;
    this.#departamentos = [];
  }

  get nome() { return this.#nome; }

  adicionarDepartamento(departamento) {
    this.#departamentos.push(departamento);
  }

  listarDepartamentos() {
    return this.#departamentos.map(dep => dep.nome);
  }
}
Exemplo de Uso
// Criando departamentos
const ti = new Departamento("TI");
const rh = new Departamento("RH");
const financas = new Departamento("Finanças");

// Criando empresa
const empresa = new Empresa("IFMG");

// Adicionando departamentos
empresa.adicionarDepartamento(ti);
empresa.adicionarDepartamento(rh);
empresa.adicionarDepartamento(financas);

// Listando departamentos
console.log(empresa.listarDepartamentos());
// Output: ["TI", "RH", "Finanças"]
